class Solution:
    def largeGroupPositions(self, S: str) -> List[List[int]]:
        res = []
        precur = 0
        for i, c in enumerate(S):
            if c == S[precur]:
                continue
            if i - precur > 2:
                res += [precur, i - 1], 
            precur = i
        if len(S) - precur > 2:
            res += [precur, len(S) - 1], 
        return res
