#### Code
```python
class Solution:
    def licenseKeyFormatting(self, S: str, K: int) -> str:
        chars = [c.upper() for s in S.split('-') for c in s]
        res = []
        for i in range(len(chars), 0, -K):
            res.append(''.join(chars[max(i-K, 0):i]))
        res.reverse()
        return '-'.join(res)
```
