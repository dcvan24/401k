```python
class Solution:
    def isValidSudoku(self, board: List[List[str]]) -> bool:
        from collections import defaultdict 
        
        m, n = len(board), len(board[0])
        col, grid = defaultdict(set), defaultdict(set)
        for i in range(m):
            row = set()
            for j in range(n):
                num = board[i][j]
                if num == '.':
                    continue
                gid = i//3, j//3
                if num in row or num in col[j] or num in grid[gid]:
                    return False
                row.add(num)
                col[j].add(num)
                grid[gid].add(num)
                
        return True
```
