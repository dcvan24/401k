```python
class Solution:
    def convertToBase7(self, num: int) -> str:
        if num == 0:
            return '0'
        res, neg = '', num < 0
        if num < 0: 
            num = -num
        while num != 0:
            res = '%d%s'%(num%7, res)
            num //= 7
        return '-%s'%res if neg else res
```
