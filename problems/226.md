226 Invert Binary Tree

#### Code
```python
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def invertTree(self, root: TreeNode) -> TreeNode:
        
        def traverse(node):
            if not node:
                return node
            l, r = node.left, node.right
            node.left = traverse(r)
            node.right = traverse(l)
            return node
        
        return traverse(root)
```
