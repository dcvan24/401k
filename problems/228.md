##### Python
```python
class Solution:
    def summaryRanges(self, nums: List[int]) -> List[str]:
        res = []
        if not nums:
            return res
        start, pre = -1, -1
        for i, n in enumerate(nums):
            if i == 0:
                start = pre = n
                continue
            if n - pre == 1:
                pre = n
                continue
            res += ('%d->%d'%(start, nums[i - 1]) if start < nums[i - 1] else '%d'%start),
            start = pre = n
        res += ('%d->%d'%(start, nums[-1]) if start < nums[-1] else '%d'%start),
        return res
                
```

##### Go 
```go
import "fmt"

func summaryRanges(nums []int) []string {
    var (
        res []string
        start, pre int
        r string
    )
    
    if len(nums) == 0 {
        return res
    }
    
    for i, n := range nums {
        if i == 0 {
            start, pre = n, n
        } else if n - pre == 1 {
            pre = n
        } else {
            if pre == start {
                r = fmt.Sprintf("%d", start)
            } else {
                r = fmt.Sprintf("%d->%d", start, pre)
            }
            res = append(res, r)
            start, pre = n, n
        }
    }
    
    if pre == start {
        r = fmt.Sprintf("%d", start)
    } else {
        r = fmt.Sprintf("%d->%d", start, pre)
    }
    res = append(res, r)
    
    
    return res
}
```
