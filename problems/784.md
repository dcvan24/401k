784 Letter Case Permutation

```python
class Solution:
    def letterCasePermutation(self, S: str) -> List[str]:
        res = []
        
        from collections import deque
        q = deque(['',])
        while q:
            prefix = q.popleft()
            if len(prefix) == len(S):
                res.append(prefix)
                continue
            c = S[len(prefix)]

            q.append(prefix + c)
            if c.isdigit():
                continue
            if c.isupper():
                q.append(prefix + c.lower())
            else:
                q.append(prefix + c.upper())
                
        return res
```
